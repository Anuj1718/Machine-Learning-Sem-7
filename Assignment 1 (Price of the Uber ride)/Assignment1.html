# Haversine distance function
def haversine_distance(lat1, lon1, lat2, lon2):
    R = 6371  # Earth's radius in km
    phi1 = np.radians(lat1)
    phi2 = np.radians(lat2)
    d_phi = np.radians(lat2 - lat1)
    d_lambda = np.radians(lon2 - lon1)

    a = np.sin(d_phi / 2.0)**2 + \
        np.cos(phi1) * np.cos(phi2) * np.sin(d_lambda / 2.0)**2

    c = 2 * np.arcsin(np.sqrt(a))
    return R * c

# Apply Haversine formula to get distance
df['trip_distance_km'] = haversine_distance(
    df['pickup_latitude'],
    df['pickup_longitude'],
    df['dropoff_latitude'],
    df['dropoff_longitude']
)

# 2. Outlier detection using IQR
def detect_outliers_iqr(data, column):
    Q1 = data[column].quantile(0.25)
    Q3 = data[column].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    outliers = data[(data[column] < lower_bound) | (data[column] > upper_bound)]

    print(f"\nOutliers in '{column}':")
    print(f"Q1 = {Q1:.2f}, Q3 = {Q3:.2f}, IQR = {IQR:.2f}")
    print(f"Lower Bound = {lower_bound:.2f}, Upper Bound = {upper_bound:.2f}")
    print(f"Number of Outliers: {len(outliers)}")

    # Boxplot
    plt.figure(figsize=(6, 4))
    sns.boxplot(x=data[column])
    plt.title(f'Boxplot for {column}')
    plt.show()

    return outliers

outlier_columns = ['fare_amount', 'trip_distance_km', 'passenger_count']

def remove_outliers_iqr(data, column):
    Q1 = data[column].quantile(0.25)
    Q3 = data[column].quantile(0.75)
    IQR = Q3 - Q1
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR
    return data[(data[column] >= lower) & (data[column] <= upper)]

# Apply to all outlier columns

# 7. Evaluation function
def evaluate_model(y_true, y_pred, model_name):
    r2 = r2_score(y_true, y_pred)
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    print(f"\n{model_name} Evaluation:")
    print(f"R Score: {r2}")
    print(f"RMSE: {rmse}")

# Ideal R and RMSE (for Uber fare prediction)
#  R Score (Coefficient of Determination)
# R  1.0  Perfect prediction (model explains nearly all variance)

# R > 0.95  Excellent (usually with very clean or simulated data)

# R  0.800.90  Good, especially for real-world noisy data like Uber

# R < 0.70  Likely too much noise, missing key variables, or not enough filtering

# Goal: For your project with filtering, aim for R  0.90  this means your model is closely predicting actual fares.

#  RMSE (Root Mean Squared Error)
# This is in the same unit as fare_amount (USD).

# Ideal RMSE depends on your fare range.

# Fare Range	Ideal RMSE
# $5$60 (your case)	RMSE < $3.00
# Narrower data	RMSE < $2.00

# Goal: Try to get RMSE below $3, ideally close to $2 or less, which means average prediction error is small.
